function [P] = LagrangeInterpolation(X, x, y)
n = length(x);
P = 0;
for i = 1:n
Pj = y(i);
for j = 1:n
if i ~= j
Pj = Pj .* ((X-x(j))/(x(i)-x(j)));
end
end
P = P + Pj;
end
P = simplify(vpa(P));
end
function [L, U] = LU_decomposition(A)
n = size(A,2);
U = A;
L = zeros(size(A));
for i = 1:n
for j = 1:i-1
L(i,j) = U(i,j);
U(i,:) = U(i,:) - U(i,j)*U(j,:);
end
L(i,i) = U(i,i);
U(i,:) = U(i,:) / U(i,i);
end
end
function [L, U] = LU_decomposition_UH(A)
n = size(A,2);
U = A;
L = zeros(size(A));
for i = 1:n
if i>1
L(i,i-1) = U(i,i-1);
U(i,:) = U(i,:) - U(i,i-1)*U(i-1,:);
end
L(i,i) = U(i,i);
U(i,:) = U(i,:) / U(i,i);
end
end
